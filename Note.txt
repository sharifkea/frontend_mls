def encrypt_and_send_message(group_id_b64: str, message_text: str, token: str, user_id: str, group_state: dict):
    """
    Encrypt an application message using current epoch_secret and send to server
    """
    try:
        #if user_id not in user_crypto_store or group_id_b64 not in user_crypto_store[user_id].get('groups', {}):
            #return {"error": "Not in group or no group state"}

        #group_state = user_crypto_store[user_id]['groups'][group_id_b64]
        
        epoch = group_state['group_last_epoch']
        epoch_secret = group_state['epoch_secret']  # Decode from base64
        my_leaf_index = group_state['my_leaf_index']
        cipher_suite = group_state['cipher_suite']
        group_id_bytes = base64.b64decode(group_id_b64)
        
        print(f"🔐 Encrypting message for group {group_id_b64} at epoch {epoch}")

        # 1. Create FramedContent (the content to be encrypted)
        from mls_stuff.MLS import FramedContent, Sender
        from mls_stuff.Enums import SenderType, ContentType
        from mls_stuff.Misc import VLBytes
        
        sender = Sender(sender_type=SenderType.member, leaf_index=my_leaf_index)
        
        framed_content = FramedContent(
            group_id=VLBytes(group_id_bytes),
            epoch=epoch,
            sender=sender,
            authenticated_data=VLBytes(b""),
            content_type=ContentType.application,
            application_data=VLBytes(message_text.encode('utf-8'))
        )
        print("###################### 1. is done. #########################")
        # 2. Serialize the content to be encrypted
        content_bytes = framed_content.serialize()
        print("###################### 2. is done. #########################")

        # 3. Derive message encryption key (MLS spec: DeriveSecret(epoch_secret, "message key"))
        from mls_stuff.Crypto._derive_secrets import DeriveSecret
        message_key = DeriveSecret(cipher_suite, epoch_secret, b"message key")
        print("###################### 3. is done. #########################")
        
        # 4. Generate random nonce (12 bytes for AES-GCM)
        import secrets
        from cryptography.hazmat.primitives.ciphers.aead import AESGCM
        
        nonce = secrets.token_bytes(12)
        print("###################### 4. is done. #########################")

        # 5. Encrypt the FramedContent
        aead = AESGCM(message_key)
        ciphertext = aead.encrypt(nonce, content_bytes, b"")
        print("###################### 5. is done. #########################")

        # 6. Create PrivateMessage (MLS wire format)
        from mls_stuff.MLS import PrivateMessage, MLSMessage
        from mls_stuff.Enums import WireFormat
        
        private_msg = PrivateMessage(
            group_id=VLBytes(group_id_bytes),
            epoch=epoch,
            content_type=ContentType.application,
            authenticated_data=VLBytes(b""),
            encrypted_sender_data=VLBytes(b""),  # Simplified for now
            ciphertext=VLBytes(ciphertext)
        )
        print("###################### 6. is done. #########################")

        # 7. Wrap in MLSMessage
        mls_message = MLSMessage(
            wire_format=WireFormat.MLS_PRIVATE_MESSAGE,
            msg_content=private_msg
        )
        print("###################### 7. is done. #########################")

        # 8. Serialize and send to server
        serialized = mls_message.serialize()
        
        payload = {
            "group_id": group_id_b64,
            "ciphertext": base64.b64encode(serialized).decode('ascii'),
            "nonce": base64.b64encode(nonce).decode('ascii'),
            "epoch": epoch,
            "content_type": 1,  # application
            "wire_format": 2    # private message
        }
        
        response = requests.post(
            f"{BASE_URL}/messages",
            json=payload,
            headers={"Authorization": f"Bearer {token}"}
        )
        
        if response.status_code == 200:
            print(f"✅ Message sent successfully")
            return {"success": True, "message": "Message sent"}
        else:
            return {"error": f"Server error: {response.text}"}

    except Exception as e:
        import traceback
        traceback.print_exc()
        return {"error": str(e)}

def derive_epoch_secret_from_tree(tree: RatchetTree, cipher_suite: CipherSuite) -> bytes:
    """Derive epoch secret from a properly repaired tree"""
    if tree is None:
        raise ValueError("No tree provided")
    
    print(f"Deriving epoch secret from tree with {len(tree.leaves)} leaves")
    
    # Final sanity check - ensure ALL leaves have _leaf_index
    for i, leaf in enumerate(tree.leaves):
        if isinstance(leaf, LeafNode):
            if not hasattr(leaf, '_leaf_index') or leaf._leaf_index is None:
                leaf._leaf_index = i
                print(f"  Forced leaf index {i} on leaf")
    
    # Also ensure node indices
    for i in range(tree.nodes):
        node = tree[i]
        if hasattr(node, '_node_index'):
            if node._node_index is None:
                node._node_index = i
    
    tree.update_leaf_index()
    tree.update_node_index()
    
    # Now hash should work
    try:
        root_secret = tree.hash(cipher_suite)
        print(f"   root_secret (first 8): {root_secret[:8].hex()}")
        print(f"✅ tree.hash() succeeded")
    except Exception as e:
        print(f"❌ tree.hash() failed: {e}")
        # Last resort - manual hash calculation
        raise
    
    epoch_secret = DeriveSecret(cipher_suite, root_secret, b"epoch")
    print(f"   epoch_secret (first 8): {epoch_secret[:8].hex()}")
    return epoch_secret
